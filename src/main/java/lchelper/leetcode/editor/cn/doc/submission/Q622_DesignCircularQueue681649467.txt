class MyCircularQueue {
    // 用循环数组，数组长度多一个，为k+1，容易区分是空还是满
    private int[] ele;
    private int start;
    private int end;
    // 数组长度
    private int cap;

    public MyCircularQueue(int k) {
        cap = k + 1;
        ele = new int[cap];
        start = 0;
        end = 0;
    }
    
    public boolean enQueue(int value) {
        if (isFull()) {
            return false;
        }
        ele[end] = value;
        end = (end + 1) % cap;
        return true;
    }
    
    public boolean deQueue() {
        if (isEmpty()) {
            return false;
        }
        start = (start + 1) % cap;
        return true;
    }
    
    public int Front() {
        return isEmpty() ? -1 : ele[start];
    }
    
    public int Rear() {
        return isEmpty() ? -1 : ele[(end - 1 + cap) % cap];
    }
    
    public boolean isEmpty() {
        return start == end;
    }
    
    public boolean isFull() {
        return start == (end + 1) % cap;
    }
}

 * int param_3 = obj.Front();
 * int param_4 = obj.Rear();
 * boolean param_5 = obj.isEmpty();
 * boolean param_6 = obj.isFull();
